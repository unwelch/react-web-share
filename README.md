## Install

```bash
npm install --save react-web-share-api-polyfill
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-web-share-api-polyfill'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
